//平铺转树
const arrayToTree = (data) => {
  let map = new Map();
  let result = [];
  data.forEach((item) => map.set(item['id'], { ...item, children: [] }));

  for (let { id, pid } of data) {
    var currentItem = map.get(id);
    parentItem = map.get(pid);

    pid == 0
      ? result.push(currentItem)
      : parentItem['children'].push(currentItem);
  }

  return result;
};
