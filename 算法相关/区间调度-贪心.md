//区间调度算法
function intervalSchedule(arr) {
  if (arr.length === 0) return 0;
  const sortArr = arr.sort((a, b) => a[1] - b[1]);
  let count = 1;
  let scheduleOfEnd = sortArr[0][1];

  for (let i = 0; i < arr.length; i++) {
    if (sortArr[i][0] >= scheduleOfEnd) {
      scheduleOfEnd = sortArr[i][1];
      count++;
    }
  }
  return count;
}

let timeId = null;
const mockSetTimeOut = (fn, delay, thisArg) => {
  (function outPut() {
    timeId = setTimeout(() => {
      fn.call(thisArg, thisArg);
      outPut();
    }, delay);
  })();
};

const cancelMockSetTimeOut = (id) => clearTimeout(id);

mockSetTimeOut((obj) => console.log(obj.name), 1000, { name: 'sxy' });
