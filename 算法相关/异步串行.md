res为异步任务队列，需要依次顺序执行。

```js
let res = [
  () => ({ success: true, data: 3000 }),
  (res) =>
    new Promise((resolve) =>
      setTimeout(
        () => resolve({ success: true, data: res + '   ' + 'data1' }),
        1000
      )
    ),
  (res) =>
    new Promise((resolve) =>
      setTimeout(
        () => resolve({ success: true, data: res + '   ' + 'data2' }),
        1000
      )
    )
];

async function foo(arr) {
  return new Promise(async (resolve, reject) => {
    let result = null;
    for (let task of arr) {
      let { success, data } = await task(result);
      result = data;
    }
    resolve(result);
  });
}

foo(res).then(console.log);
```
